Câu 22.

Câu 23.
SELECT KHACHHANG.MAKH
FROM HOADON JOIN KHACHHANG ON HOADON.MAKH = KHACHHANG.MAKH
WHERE  HOADON.TRIGIA =
(
    SELECT MAX(HOADON.TRIGIA)
    FROM HOADON
    WHERE HOADON.NGHD LIKE "%2006"
)
